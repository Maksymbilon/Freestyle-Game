  <script>
    const jumper = document.getElementById('jumper');
    const result = document.getElementById('result');

    let jumping = false;

    jumper.addEventListener('click', () => {
      if (jumping) return;
      jumping = true;

      const height = Math.floor(Math.random() * 30) + 10; // random jump height 10â€“40m
      jumper.style.transform = `translateY(-${height * 4}px)`;

      setTimeout(() => {
        result.textContent = `Wow! You reached ${height} meters! Ready to support real athletes in Ukraine?`;
        result.classList.remove('hidden');
        jumping = false;
      }, 500);
    });

    function resetJump() {
      jumper.style.transform = 'translateY(0)';
      result.classList.add('hidden');
    }
  </script>
